def sum1(n):
    total = 0
    for i in range(1, n + 1):
        total += 1 / float(i)
    return total
def sum2(n):
    total = 0
    for i in range(n ,0 ,-1):
        total += 1 / float(i)
    return total

n = 100000
result = sum1(n)
print(f"sum1 for n={n} is {result}")
result = sum2(n)
print(f"sum2 for n={n} is {result}")
print(f"sum1(n) == sum2(n) is {sum1(n) == sum2(n)} \n sum1 , sum2 = {sum1(n)} , {sum2(n)} \n sum1 - sum2 = {sum2(n) - sum1(n)}")
